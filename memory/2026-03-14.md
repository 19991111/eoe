self.sources[i, 4] -= inject_amount  # 可能变负

# 修复后
remaining = self.sources[i, 4].item()
if remaining > 0:
    actual_inject = min(inject_amount, remaining)
    self.field[0, 0, gy, gx] += actual_inject
    self.sources[i, 4] -= actual_inject

    self.sources[i, 4] = 0  # 防止负数
